147static void generateSphereMesh(std::vector<Position>& vertices, std::vector<GLuint>& indices,
148 float radius = 1.0f, unsigned int sectorCount = 36, unsigned int stackCount = 18) {
149 const float PI = 3.14159265359f;
150
151 vertices.clear();
152 indices.clear();
153
154 for (unsigned int i = 0; i <= stackCount; ++i) {
155 float stackAngle = PI / 2.0f - i * PI / stackCount; // from pi/2 to -pi/2
156 float xy = radius * cosf(stackAngle);
157 float z = radius * sinf(stackAngle);
158
159 for (unsigned int j = 0; j <= sectorCount; ++j) {
160 float sectorAngle = j * 2.0f * PI / sectorCount;
161
162 float x = xy * cosf(sectorAngle);
163 float y = xy * sinf(sectorAngle);
164 glm::vec3 pos(x, y, z);
165
166 vertices.push_back(pos);
167 }
168 }
169
170 for (unsigned int i = 0; i < stackCount; ++i) {
171 unsigned int k1 = i * (sectorCount + 1);
172 unsigned int k2 = k1 + sectorCount + 1;
173
174 for (unsigned int j = 0; j < sectorCount; ++j, ++k1, ++k2) {
175 if (i != 0) {
176 indices.push_back(k1);
177 indices.push_back(k2);
178 indices.push_back(k1 + 1);
179 }
180 if (i != (stackCount - 1)) {
181 indices.push_back(k1 + 1);
182 indices.push_back(k2);
183 indices.push_back(k2 + 1);
184 }
185 }
186 }
187}

342 void compileShadersFromSource(const char* vertexSource, const char* fragmentSource) {
343
344 // Vertex and fragment shaders are successfully compiled.
345 // Now time to link them together into a program.
346 // Get a program object.
347 _programID = glCreateProgram();
348
349 _vertexShaderID = glCreateShader(GL_VERTEX_SHADER);
350 if (_vertexShaderID == 0) {
351 TazGraphEngine::ConsoleLogger::error("Vertex Shader Failed to create!");
352 }
353
354 _fragmentShaderID = glCreateShader(GL_FRAGMENT_SHADER);
355 if (_fragmentShaderID == 0) {
356 TazGraphEngine::ConsoleLogger::error("Fragment Shader Failed to create!");
357 }
358
359 compileShader(vertexSource, "vertex Shader", _vertexShaderID);
360 compileShader(fragmentSource, "fragment Shader", _fragmentShaderID);
361 }
362
363 void linkShaders() {
364 // Attach our shaders to our program
365 glAttachShader(_programID, _vertexShaderID);
366 glAttachShader(_programID, _fragmentShaderID);
367
368 // Link our program
369 glLinkProgram(_programID);
370
371 // Note the different functions here: glGetProgram* instead of glGetShader*.
372 GLint isLinked = 0;
373 glGetProgramiv(_programID, GL_LINK_STATUS, (int*)&isLinked);
374 if (isLinked == GL_FALSE)
375 {
376 GLint maxLength = 0;
377 glGetProgramiv(_programID, GL_INFO_LOG_LENGTH, &maxLength);
378
379 // The maxLength includes the NULL character
380 std::vector<GLchar> errorLog(maxLength);
381 glGetProgramInfoLog(_programID, maxLength, &maxLength, &errorLog[0]);
382
383 // We don't need the program anymore.
384 glDeleteProgram(_programID);
385 // Don't leak shaders either.
386 glDeleteShader(_vertexShaderID);
387 glDeleteShader(_fragmentShaderID);
388
389 std::printf("%s\n", &(errorLog[0]));
390 TazGraphEngine::ConsoleLogger::error("Shaders failed to link");
391 }
392
393 // Always detach shaders after a successful link.
394 glDetachShader(_programID, _vertexShaderID);
395 glDetachShader(_programID, _fragmentShaderID);
396 glDeleteShader(_vertexShaderID);
397 glDeleteShader(_fragmentShaderID);
398 }
